		window_name = "government_main_container"
		visible = {

		}
		effects= {
			leader_frame2_click = {
				if = {

				}

					set_variable = { government_button_open = 1 }
				}
			}
		}
	}

		context_type = player_context
		window_name = "government_main_window"
		visible = {
			check_variable = { government_button_open = 1 }
		}
